# コピーコマンド

```sh

./app
└─temple
    ├─sample01
    | └─page.tsx
    ├─sample02
    |  └─page.tsx    
    ├─sample03
    | └─page.tsx
    └─sample04
      └─page.tsx
```

cp -n --parents app/temple/sample02/page.tsx /home/temp/
cp -n --parents app/temple/sample03/page.tsx /homp/temp/
