# markdownファイルを表示させる方法

## ライブラリーインポート
```sh
## Markdownレンダリング用ライブラリ
npm install react-markdown @tailwindcss/typography remark-gfm rehype-highlight
```

## globals.cssに追加
- [参考]https://github.com/tailwindlabs/tailwindcss-typography
```css
@plugin "@tailwindcss/typography";
```

## 実装例

- コンポーネント
- src/components/parts/MarkdownParts.tsx
```tsx
import Markdown from 'react-markdown';
import rehypeHighlight from "rehype-highlight";
import remarkGfm from 'remark-gfm';
// propsの定義
type Props = { content: string };
export default function MarkdownParts({ content }: Props) {
  return (
    <div className="prose max-w-none">
      <Markdown
        remarkPlugins={[remarkGfm]}
        rehypePlugins={[rehypeHighlight]}
        components={{
          a: ({ ...props }) => (
            <a {...props}
              target="_blank"
              rel="noopener noreferrer"
              className="text-blue-600 dark:text-blue-400 hover:underline"
            >
              {props.children}
            </a>
          ),
          p: ({ ...props }) => (
            <p {...props}
              className="mb-4 leading-7"
            >
              {props.children}
            </p>
          ),
          ul: ({ ...props }) => (
            <ul {...props}
              className="mb-4 ml-6 list-disc space-y-1"
            >
              {props.children}
            </ul>
          ),
          ol: ({ ...props }) => (
            <ol {...props}
              className="mb-4 ml-6 list-decimal space-y-1"
            >
              {props.children}
            </ol>
          ),
          li: ({ ...props }) => (
            <li {...props}
              className="leading-7"
            >
              {props.children}
            </li>
          ),
          table: ({ ...props }) => (
            <div className="mb-4 overflow-x-auto">
              <table className="min-w-full border-collapse border border-gray-300 dark:border-gray-700" {...props}>
                {props.children}
              </table>
            </div>
          ),
          thead: ({ ...props }) => (
            <thead className="bg-gray-100 dark:bg-gray-800" {...props}>
              {props.children}
            </thead>
          ),
          th: ({ ...props }) => (
            <th className="px-4 py-2 text-left border border-gray-300 dark:border-gray-700 font-semibold" {...props}>
              {props.children}
            </th>
          ),
          td: ({ ...props }) => (
            <td className="px-4 py-2 border border-gray-300 dark:border-gray-700" {...props}>
              {props.children}
            </td>
          ),
          }}
        >
        {content}
      </Markdown>
    </div>
  )
}
```

- markdownファイル読み込み

```tsx
import fs from "fs";
import path from "path";
import MarkdownParts from "@/components/parts/markdown";

// markdownfile読み込み
const files = ["sample01.md","sample02.md"];
const contents = files.map((file) => {
  const filePath = path.join(process.cwd(), "public", "md", file);
  return fs.readFileSync(filePath, "utf-8");
});

export default function SamplePage() {
  return (
    <main>
      // sample01.md
      <MarkdownParts content={contents[0]} />
      // sample02.md
      <MarkdownParts content={contents[1]} />
    </main>
  )
}
```
